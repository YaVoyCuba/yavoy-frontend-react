# Internationalization (i18n) Rules

Load this when working with translations, multi-language support, or locale-specific formatting.

## Setup

### Use Lingui or react-intl
```ts
// Good: Lingui (recommended for TypeScript)
import { i18n } from '@lingui/core';
import { useLingui } from '@lingui/react';

i18n.activate('en');
```

## Translation Extraction

### Mark strings for translation
```tsx
// Good: Use t macro for static strings
import { Trans, useLingui } from '@lingui/react';

export function OrderDetail() {
  const { i18n } = useLingui();
  
  return (
    <div>
      <Trans>Order Details</Trans>
      <p>{i18n._(t`Total: ${total}`)}</p> {/* Dynamic content */}
    </div>
  );
}

// Avoid: Hardcoded strings
return <div>Order Details</div>; // Won't be translated
```

### Pluralization
```ts
// Good: Handle plural forms correctly
import { Trans } from '@lingui/react';

<Trans>
  {count, plural, one {# item} other {# items}}
</Trans>
```

## Locale-Specific Formatting

### Date formatting
```ts
// Good: Use Intl API
const date = new Date();
const formatted = new Intl.DateTimeFormat('es-ES', {
  year: 'numeric',
  month: 'long',
  day: 'numeric',
}).format(date); // "8 de junio de 2024"
```

### Currency formatting
```ts
// Good: Locale-aware currency
const price = new Intl.NumberFormat('es-ES', {
  style: 'currency',
  currency: 'EUR',
}).format(99.99); // "99,99 €"
```

### Number formatting
```ts
// Good: Locale-aware numbers
const number = new Intl.NumberFormat('de-DE').format(1234567.89); // "1.234.567,89"
```

## URL locale routing

### Locale in URL path
```ts
// Good: Route with locale
<BrowserRouter basename={`/${locale}`}>
  <Routes>
    <Route path="/orders" element={<Orders />} />
  </Routes>
</BrowserRouter>

// URLs: /en/orders, /es/orders, /fr/orders
```

### Detect locale from URL
```ts
// Good: Extract locale from pathname
export function useLocale(): string {
  const pathname = useLocation().pathname;
  const [locale] = pathname.split('/').filter(Boolean);
  return locale || 'en';
}
```

## Language Switcher

```tsx
// Good: Allow user to switch language
export function LanguageSwitcher() {
  const navigate = useNavigate();
  const currentLocale = useLocale();
  
  const handleChange = (newLocale: string) => {
    const pathname = window.location.pathname;
    const newPathname = pathname.replace(`/${currentLocale}`, `/${newLocale}`);
    navigate(newPathname);
    document.documentElement.lang = newLocale;
  };
  
  return (
    <select value={currentLocale} onChange={(e) => handleChange(e.currentTarget.value)}>
      <option value="en">English</option>
      <option value="es">Español</option>
      <option value="fr">Français</option>
    </select>
  );
}
```
